60  ! (節点1->節点2) x (節点1->節点3)を計算
61  self%an(:,i) = vector_product(self%p(:,self%nd(2,i))-self%p(:,self%nd(1,i)), self%p(:,self%nd(3,i))-self%p(:,self%nd(1,i)))
62  ! 面積 = ↑の外積の長さ / 2
