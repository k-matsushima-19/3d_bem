80  subroutine init(self)
81  class(mesh3d),intent(inout) :: self
82 
83  integer :: i
84  real(8) :: ave, var
85 
86  call assert(allocated(self%p))
87  call assert(allocated(self%nd))
88 
89  ! 重心
90  if(allocated(self%c)) deallocate(self%c)
91  allocate(self%c(3,self%ne))
92  do i=1,self%ne
93  self%c(:,i) = (self%p(:,self%nd(1,i))+self%p(:,self%nd(2,i))+self%p(:,self%nd(3,i))) / 3.d0
94  end do
95 
96  ! 法線と要素面積
97  if(allocated(self%an)) deallocate(self%an)
98  if(allocated(self%ar)) deallocate(self%ar)
99  allocate(self%an(3,self%ne))
100  allocate(self%ar(self%ne))
101  do i=1,self%ne
102  ! (節点1->節点2) x (節点1->節点3)を計算
103  self%an(:,i) = vector_product(self%p(:,self%nd(2,i))-self%p(:,self%nd(1,i)), self%p(:,self%nd(3,i))-self%p(:,self%nd(1,i)))
104  ! 面積 = ↑の外積の長さ / 2
105  self%ar(i) = vector_length(self%an(:,i)) * 0.5d0
106  ! 単位ベクトルにする
107  self%an(:,i) = self%an(:,i) / (self%ar(i)*2.d0)
108  end do
109 
110  call self%calc_enclosing_circle(self%radius, self%center)
111 
112  ! 要素面積の平均
113  ave = sum(self%ar) / self%ne
114  ! 要素面積の分散
115  var = sum((self%ar-ave)**2) / self%ne
116 
117  write(*,*) "# Number of elements: ", self%ne
118  write(*,*) "# Number of vertices: ", self%np
119  write(*,*) "# Minimum area of triangles: ", minval(self%ar)
120  write(*,*) "# Maximum area of triangles: ", maxval(self%ar)
121  write(*,*) "# CV of triangle areas: ", sqrt(var) / ave
122 
123 
124  end subroutine init
