64  subroutine init(self)
65  class(mesh3d),intent(inout) :: self
66 
67  integer :: i
68  real(8) :: ave, var
69 
70  call assert(allocated(self%p))
71  call assert(allocated(self%nd))
72 
73  ! 重心
74  if(allocated(self%c)) deallocate(self%c)
75  allocate(self%c(3,self%ne))
76  do i=1,self%ne
77  self%c(:,i) = (self%p(:,self%nd(1,i))+self%p(:,self%nd(2,i))+self%p(:,self%nd(3,i))) / 3.d0
78  end do
79 
80  ! 法線と要素面積
81  if(allocated(self%an)) deallocate(self%an)
82  if(allocated(self%ar)) deallocate(self%ar)
83  allocate(self%an(3,self%ne))
84  allocate(self%ar(self%ne))
85  do i=1,self%ne
86  ! (節点1->節点2) x (節点1->節点3)を計算
87  self%an(:,i) = vector_product(self%p(:,self%nd(2,i))-self%p(:,self%nd(1,i)), self%p(:,self%nd(3,i))-self%p(:,self%nd(1,i)))
88  ! 面積 = ↑の外積の長さ / 2
89  self%ar(i) = vector_length(self%an(:,i)) * 0.5d0
90  ! 単位ベクトルにする
91  self%an(:,i) = self%an(:,i) / (self%ar(i)*2.d0)
92  end do
93 
94  call self%calc_enclosing_circle(self%radius, self%center)
95 
96  ! 要素面積の平均
97  ave = sum(self%ar) / self%ne
98  ! 要素面積の分散
99  var = sum((self%ar-ave)**2) / self%ne
100 
101  write(*,*) "# Number of elements: ", self%ne
102  write(*,*) "# Number of vertices: ", self%np
103  write(*,*) "# Minimum area of triangles: ", minval(self%ar)
104  write(*,*) "# Maximum area of triangles: ", maxval(self%ar)
105  write(*,*) "# CV of triangle areas: ", sqrt(var) / ave
106 
107 
108  end subroutine init
