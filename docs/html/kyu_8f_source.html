341 c***********************************************************************
342  implicit none
343  integer nmax,n
344  real*8 rad,s(3),e(3),v(3)
345 *
346  integer i

349 c
350  if(nmax.eq.0)then ! s=e=v
351  do i=1,3
352  v(i)=s(i)
353  enddo
354  else
355  ds=dsqrt(s(1)**2+s(2)**2+s(3)**2)
356  de=dsqrt(e(1)**2+e(2)**2+e(3)**2)
357  do i=1,3
358  vs(i)=s(i)/ds
359  ve(i)=e(i)/de
360  enddo
361  vn(1)=vs(2)*ve(3)-vs(3)*ve(2)
362  vn(2)=vs(3)*ve(1)-vs(1)*ve(3)
363  vn(3)=vs(1)*ve(2)-vs(2)*ve(1)

365  do i=1,3

367  enddo

369  vp(2)=vn(3)*vs(1)-vn(1)*vs(3)
370  vp(3)=vn(1)*vs(2)-vn(2)*vs(1)
371  angle=dacos(vs(1)*ve(1)+vs(2)*ve(2)+vs(3)*ve(3))
