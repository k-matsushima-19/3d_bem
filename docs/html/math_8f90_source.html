226  subroutine sph(n,x,ynm)
227  integer,intent(in) :: n
228  real(8),intent(in) :: x(3)
229  complex(8),intent(out) :: ynm((n+1)**2)
230 
231  integer :: m, i, k, j
232  complex(8) :: cc
233  real(8) :: v(3), ctmp, xlen
234 
235  xlen=sqrt(dot_product(x,x))
236 
237  do i=1,3
238  v(i)=x(i)/xlen
239  enddo
240 
241  cc=cmplx(v(1),v(2),kind(1.d0))
242 
243  ynm(loct(0,0))=1d0
244 
245  do m=0,n-1
246  ynm(loct(m+1,m+1))=cc*ynm(loct(m,m))*sqrt((2d0*m+1d0)/(2d0*m+2d0))
247  enddo
248 
249 
250  do m=0,n-1
251  ynm(loct(m+1,m))=ynm(loct(m,m))*v(3)*sqrt(2d0*m+1d0)
252  do i=m+2,n
253  ctmp=sqrt(1d0*(i+m)*(i-m))
254  ynm(loct(i,m))=(v(3)*(2d0*i-1d0)*ynm(loct(i-1,m))-ynm(loct(i-2,m))*sqrt(1d0*(i-1+m)*(i-1-m)))/ctmp
255  enddo
256  enddo
257 
258  do k=-1,-n,-1
259  do j=-k,n
260  ynm(loct(j,k))=(-1.0)**(-k)*conjg(ynm(loct(j,-k)))
261  enddo
262  enddo
263 
264  return
265  end subroutine sph
