202  subroutine sph(n,x,ynm)
203  integer,intent(in) :: n
204  real(8),intent(in) :: x(3)
205  complex(8),intent(out) :: ynm((n+1)**2)
206 
207  integer :: m, i, k, j
208  complex(8) :: cc
209  real(8) :: v(3), ctmp, xlen
210 
211  xlen=sqrt(dot_product(x,x))
212 
213  do i=1,3
214  v(i)=x(i)/xlen
215  enddo
216 
217  cc=cmplx(v(1),v(2),kind(1.d0))
218 
219  ynm(loct_sph(0,0))=1d0
220 
221  do m=0,n-1
222  ynm(loct_sph(m+1,m+1))=cc*ynm(loct_sph(m,m))*sqrt((2d0*m+1d0)/(2d0*m+2d0))
223  enddo
224 
225 
226  do m=0,n-1
227  ynm(loct_sph(m+1,m))=ynm(loct_sph(m,m))*v(3)*sqrt(2d0*m+1d0)
228  do i=m+2,n
229  ctmp=sqrt(1d0*(i+m)*(i-m))
230  ynm(loct_sph(i,m))=(v(3)*(2d0*i-1d0)*ynm(loct_sph(i-1,m))-ynm(loct_sph(i-2,m))*sqrt(1d0*(i-1+m)*(i-1-m)))/ctmp
231  enddo
232  enddo
233 
234  do k=-1,-n,-1
235  do j=-k,n
236  ynm(loct_sph(j,k))=(-1.0)**(-k)*conjg(ynm(loct_sph(j,-k)))
237  enddo
238  enddo
239 
240  return
241  end subroutine sph
